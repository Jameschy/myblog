voteid smallint(6) 投票ID
title varchar(120) 投票标题
votetext text 投票项目 格式：(每个项目用回车隔开)

tempid smallint(6) 投票模板ID
width int(11) 查看投票窗口宽度
height int(11) 查看投票窗口高度
votenum int(11) 总投票数
ysvotename varchar(60) 投票名称
